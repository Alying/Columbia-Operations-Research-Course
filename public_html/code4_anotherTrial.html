# Optimization (Simplex)

def gradient(params):
    a0 = params[0]
    b1 = params[1]
    return np.array([-2.0*sum((y - a0 - b1*x1))/lenT, -2.0*sum((y - a0 - b1*x1)*x1)/lenT])

def objFunc1(params):
    a0 = params[0]
    b1 = params[1]
    return sum((y - a0 - b1*x1)**2)/lenT


tol  = 1e-6
# y = a0 + b1 * x1;
a0 = 1000
b1 = 1000
params = [a0, b1]
params = scipy.optimize.fmin(objFunc1, params, xtol=tol, ftol=tol, maxiter=500, maxfun=50000)
print("Simplex:")
print(params)
